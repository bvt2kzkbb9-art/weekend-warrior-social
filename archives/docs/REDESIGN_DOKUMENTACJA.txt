COLORS:
- Background: var(--bg-card), var(--bg-elevated)

BORDER RADIUS:
- Cards: var(--r-lg) = 8px
- Medium: var(--r-md) = 6px
- Buttons: var(--r-full) = 9999px

TRANSITIONS:
- All: var(--t) = 0.2s ease-out
- Hover: color change + slight background
